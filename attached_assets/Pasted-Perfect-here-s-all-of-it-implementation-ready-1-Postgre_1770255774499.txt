## 1) Postgres schema (DDL)

```sql
-- SkillBook Postgres DDL (baseline)
-- Assumes: Postgres 15+, uuid-ossp or gen_random_uuid via pgcrypto
CREATE EXTENSION IF NOT EXISTS pgcrypto;

-- ---------- enums ----------
DO $$ BEGIN
  CREATE TYPE visibility_t AS ENUM ('public','private','internal');
EXCEPTION WHEN duplicate_object THEN NULL; END $$;

DO $$ BEGIN
  CREATE TYPE repo_role_t AS ENUM ('owner','admin','write','read');
EXCEPTION WHEN duplicate_object THEN NULL; END $$;

DO $$ BEGIN
  CREATE TYPE pr_status_t AS ENUM ('open','closed','merged');
EXCEPTION WHEN duplicate_object THEN NULL; END $$;

DO $$ BEGIN
  CREATE TYPE validation_status_t AS ENUM ('queued','running','passed','failed','canceled');
EXCEPTION WHEN duplicate_object THEN NULL; END $$;

DO $$ BEGIN
  CREATE TYPE release_status_t AS ENUM ('draft','published','yanked');
EXCEPTION WHEN duplicate_object THEN NULL; END $$;

DO $$ BEGIN
  CREATE TYPE install_status_t AS ENUM ('stable','deprecated','yanked');
EXCEPTION WHEN duplicate_object THEN NULL; END $$;

-- ---------- users / orgs ----------
CREATE TABLE IF NOT EXISTS users (
  id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  handle          text UNIQUE NOT NULL CHECK (length(handle) >= 2),
  email           text UNIQUE NOT NULL,
  display_name    text,
  avatar_url      text,
  is_verified     boolean NOT NULL DEFAULT false,
  created_at      timestamptz NOT NULL DEFAULT now(),
  updated_at      timestamptz NOT NULL DEFAULT now()
);

CREATE TABLE IF NOT EXISTS orgs (
  id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  slug            text UNIQUE NOT NULL CHECK (length(slug) >= 2),
  name            text NOT NULL,
  owner_user_id   uuid NOT NULL REFERENCES users(id) ON DELETE RESTRICT,
  created_at      timestamptz NOT NULL DEFAULT now(),
  updated_at      timestamptz NOT NULL DEFAULT now()
);

CREATE TABLE IF NOT EXISTS org_members (
  org_id          uuid NOT NULL REFERENCES orgs(id) ON DELETE CASCADE,
  user_id         uuid NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  role            repo_role_t NOT NULL DEFAULT 'read',
  created_at      timestamptz NOT NULL DEFAULT now(),
  PRIMARY KEY (org_id, user_id)
);

-- ---------- repos ----------
CREATE TABLE IF NOT EXISTS repos (
  id                uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  owner_user_id     uuid REFERENCES users(id) ON DELETE CASCADE,
  owner_org_id      uuid REFERENCES orgs(id) ON DELETE CASCADE,
  name              text NOT NULL CHECK (length(name) >= 2),
  description       text,
  visibility        visibility_t NOT NULL DEFAULT 'public',
  default_branch    text NOT NULL DEFAULT 'main',
  git_remote_url    text NOT NULL,
  git_provider      text NOT NULL DEFAULT 'gitea', -- or github/bitbucket/etc
  is_archived       boolean NOT NULL DEFAULT false,
  created_at        timestamptz NOT NULL DEFAULT now(),
  updated_at        timestamptz NOT NULL DEFAULT now(),
  CHECK (
    (owner_user_id IS NOT NULL AND owner_org_id IS NULL)
    OR
    (owner_user_id IS NULL AND owner_org_id IS NOT NULL)
  ),
  UNIQUE (owner_user_id, name),
  UNIQUE (owner_org_id, name)
);

CREATE TABLE IF NOT EXISTS repo_permissions (
  id            uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  repo_id       uuid NOT NULL REFERENCES repos(id) ON DELETE CASCADE,
  user_id       uuid NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  role          repo_role_t NOT NULL,
  created_at    timestamptz NOT NULL DEFAULT now(),
  UNIQUE (repo_id, user_id)
);

CREATE TABLE IF NOT EXISTS forks (
  id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  parent_repo_id  uuid NOT NULL REFERENCES repos(id) ON DELETE CASCADE,
  fork_repo_id    uuid NOT NULL REFERENCES repos(id) ON DELETE CASCADE,
  created_at      timestamptz NOT NULL DEFAULT now(),
  UNIQUE (parent_repo_id, fork_repo_id)
);

-- ---------- commits (indexed from git webhooks) ----------
CREATE TABLE IF NOT EXISTS commits (
  id            uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  repo_id       uuid NOT NULL REFERENCES repos(id) ON DELETE CASCADE,
  commit_hash   text NOT NULL,
  author_name   text,
  author_email  text,
  message       text,
  committed_at  timestamptz,
  created_at    timestamptz NOT NULL DEFAULT now(),
  UNIQUE (repo_id, commit_hash)
);

CREATE INDEX IF NOT EXISTS idx_commits_repo_time ON commits(repo_id, committed_at DESC);

-- ---------- artifacts (logs, tarballs, SBOM, signatures) ----------
CREATE TABLE IF NOT EXISTS artifacts (
  id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  kind            text NOT NULL,  -- e.g. "validation_log","sbom","tarball","signature"
  storage_key     text NOT NULL,  -- path/key in S3
  content_type    text NOT NULL,
  byte_size       bigint NOT NULL,
  sha256          text NOT NULL,
  created_by      uuid REFERENCES users(id) ON DELETE SET NULL,
  created_at      timestamptz NOT NULL DEFAULT now()
);

-- ---------- validations ----------
CREATE TABLE IF NOT EXISTS validations (
  id                uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  repo_id           uuid NOT NULL REFERENCES repos(id) ON DELETE CASCADE,
  commit_hash       text NOT NULL,
  status            validation_status_t NOT NULL DEFAULT 'queued',
  stage_status      jsonb NOT NULL DEFAULT '{}'::jsonb, -- {stageA:{status,...},...}
  risk_score        integer NOT NULL DEFAULT 0 CHECK (risk_score >= 0 AND risk_score <= 100),
  findings          jsonb NOT NULL DEFAULT '[]'::jsonb,  -- list of lint findings
  runner_version    text,
  started_at        timestamptz,
  finished_at       timestamptz,
  log_artifact_id   uuid REFERENCES artifacts(id) ON DELETE SET NULL,
  sbom_artifact_id  uuid REFERENCES artifacts(id) ON DELETE SET NULL,
  created_at        timestamptz NOT NULL DEFAULT now(),
  UNIQUE (repo_id, commit_hash)
);

CREATE INDEX IF NOT EXISTS idx_validations_repo_status ON validations(repo_id, status);

-- ---------- pull requests ----------
CREATE TABLE IF NOT EXISTS pull_requests (
  id               uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  repo_id          uuid NOT NULL REFERENCES repos(id) ON DELETE CASCADE,
  number           integer NOT NULL,
  title            text NOT NULL,
  body             text,
  status           pr_status_t NOT NULL DEFAULT 'open',
  source_repo_id   uuid NOT NULL REFERENCES repos(id) ON DELETE CASCADE,
  source_branch    text NOT NULL,
  target_branch    text NOT NULL,
  created_by       uuid NOT NULL REFERENCES users(id) ON DELETE RESTRICT,
  merged_by        uuid REFERENCES users(id) ON DELETE SET NULL,
  created_at       timestamptz NOT NULL DEFAULT now(),
  updated_at       timestamptz NOT NULL DEFAULT now(),
  merged_at        timestamptz,
  UNIQUE (repo_id, number)
);

-- ---------- issues ----------
CREATE TABLE IF NOT EXISTS issues (
  id           uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  repo_id      uuid NOT NULL REFERENCES repos(id) ON DELETE CASCADE,
  number       integer NOT NULL,
  title        text NOT NULL,
  body         text,
  status       text NOT NULL DEFAULT 'open', -- open/closed
  created_by   uuid NOT NULL REFERENCES users(id) ON DELETE RESTRICT,
  created_at   timestamptz NOT NULL DEFAULT now(),
  updated_at   timestamptz NOT NULL DEFAULT now(),
  UNIQUE (repo_id, number)
);

-- ---------- releases ----------
CREATE TABLE IF NOT EXISTS releases (
  id                    uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  repo_id               uuid NOT NULL REFERENCES repos(id) ON DELETE CASCADE,
  tag                   text NOT NULL,         -- e.g. v1.2.0
  version               text NOT NULL,         -- 1.2.0
  commit_hash           text NOT NULL,
  status                release_status_t NOT NULL DEFAULT 'draft',
  notes                 text,
  tarball_artifact_id   uuid REFERENCES artifacts(id) ON DELETE SET NULL,
  signature_artifact_id uuid REFERENCES artifacts(id) ON DELETE SET NULL,
  published_by          uuid REFERENCES users(id) ON DELETE SET NULL,
  published_at          timestamptz,
  created_at            timestamptz NOT NULL DEFAULT now(),
  updated_at            timestamptz NOT NULL DEFAULT now(),
  UNIQUE (repo_id, tag),
  UNIQUE (repo_id, version)
);

-- ---------- registry entries (installable) ----------
CREATE TABLE IF NOT EXISTS skill_registry_entries (
  id                 uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  repo_id            uuid NOT NULL REFERENCES repos(id) ON DELETE CASCADE,
  version            text NOT NULL,
  tag                text NOT NULL,
  commit_hash        text NOT NULL,
  manifest           jsonb NOT NULL,              -- install manifest parsed from skill.md
  integrity_sha256   text NOT NULL,
  install_status     install_status_t NOT NULL DEFAULT 'stable',
  tarball_artifact_id uuid NOT NULL REFERENCES artifacts(id) ON DELETE RESTRICT,
  signature_artifact_id uuid REFERENCES artifacts(id) ON DELETE SET NULL,
  created_at         timestamptz NOT NULL DEFAULT now(),
  UNIQUE (repo_id, version)
);

-- ---------- search cache table (optional; can be replaced by meilisearch) ----------
CREATE TABLE IF NOT EXISTS skill_search_docs (
  repo_id        uuid PRIMARY KEY REFERENCES repos(id) ON DELETE CASCADE,
  latest_version text,
  doc            jsonb NOT NULL, -- denormalized fields for indexing
  updated_at     timestamptz NOT NULL DEFAULT now()
);
```

## 3) CLI behavior + config files

### 3.1 CLI commands (contract)

**Binary:** `skillbook`

#### Auth

* `skillbook login`

  * opens browser OAuth OR prompts email/password
  * stores token in config
* `skillbook logout`
* `skillbook whoami`

#### Repo management

* `skillbook create owner/skillname [--private|--public]`

  * creates repo via API + creates remote
* `skillbook fork owner/skillname`

  * creates fork repo on server
* `skillbook init`

  * scaffolds a new skill repo in current dir:

    * creates `skill.md`, `/schemas`, `/src`, `/tests`
    * optionally runs `git init`
    * optionally creates remote repo and sets origin

#### Skill workflows

* `skillbook lint`

  * validates `skill.md` schema + references (local)
* `skillbook validate [--remote]`

  * local runs: frontmatter parse, schema, file checks, basic security grep
  * remote runs: triggers server validation
* `skillbook publish --tag vX.Y.Z [--notes file.md]`

  * checks local:

    * git clean
    * `version` in `skill.md` matches tag
  * requires server validation PASSED for commit
  * creates release (published) + registry entry
* `skillbook install owner/skill@1.2.0 [--path <opencaw_skills_dir>]`

  * downloads manifest + tarball
  * verifies sha256
  * unpacks into `<path>/<owner>/<skill>/<version>/`
  * writes lockfile
* `skillbook update owner/skill`

  * installs latest stable unless pinned
* `skillbook uninstall owner/skill`
* `skillbook list`

  * lists installed skills from lockfile

### 3.2 CLI config file locations

* **Linux/mac:** `~/.config/skillbook/config.json`
* **Windows:** `%APPDATA%\skillbook\config.json`
* Project lock: `./skillbook.lock.json` (in workspace or OpenClaw folder)
